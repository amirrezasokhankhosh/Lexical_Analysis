package sample;

public class Code{
    public static void main(String[] args){
        String input = "GeekforGeeks";
        byte[] strAsByteArray = input.getBytes();
        byte[] result = new byte[strAsByteArray.lenght];
        for(int i = 0 ; i < strAsByteArray.lenght ; i ++){
            result[i] = strAsByteArray[strAsByteArray.lenght - i - 1];
        }
        System.out.println(new String(result);
    }
}
